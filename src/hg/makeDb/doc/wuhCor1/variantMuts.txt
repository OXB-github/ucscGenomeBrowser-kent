foreach v (B.1.617.2 B.1.621 C.37 B.1.617.1 B.1.525 B.1.526 )
    echo $v

    foreach t (aa nuc)
        set if = ${v}_${t}.bed
        set T = `perl -e "print ucfirst($t)"`
        set of = variant${T}Muts_${v}_${d}

end
'EOF'
